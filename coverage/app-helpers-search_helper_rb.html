57   def thumbnail_image_tag(result, max_width=nil, max_height=nil)
 
58     width = result["Thumbnail"]["Width"].to_f
 
59     height = result["Thumbnail"]["Height"].to_f
 
60     reductions = [
 
61       (max_width && width > max_width) ? (max_width.to_f / width) : 1,
 
62       (max_height && height > max_height) ? (max_height.to_f / height) : 1
 
63     ]
 
64     reduction = reductions.min
 
65 
66     raw image_tag result["Thumbnail"]["Url"],
 
67               :width  => (width * reduction).to_i,
 
68               :height => (height * reduction).to_i,
