314   def shorten_url (url, length=30)
 
315     return url if url.length <= length
 
316     if url.count('/') >= 4
 
317       if url.index('?')
 
318         arr = url[0..url.index('?')].split('/')
 
319         arr[arr.size - 1] += url[url.index('?') + 1..-1]
 
320       else
 
321         arr = url.split('/')
 
322       end
 
323       host = arr[0]+"//"+arr[2]
 
324       if arr.last.index('?')
 
325         doc_path = arr.last.split('?')
 
326         doc = [doc_path.first, "?", doc_path[1].split('&').first, "..."].join
 
327       else
 
328         doc = arr.last.split('?').first
 
329       end
 
330       [host, "...", doc].join('/')
 
331     else
 
332       url[0, length]+"..."
 
333     end
 
334   end
