390   def shorten_url (url, length=30)
 
391     return url if url.length <= length
 
392     if url.count('/') >= 4
 
393       if url.index('?')
 
394         arr = url[0..url.index('?')].split('/')
 
395         arr[arr.size - 1] += url[url.index('?') + 1..-1]
 
396       else
 
397         arr = url.split('/')
 
398       end
 
399       host = arr[0]+"//"+arr[2]
 
400       if arr.last.index('?')

405       end
