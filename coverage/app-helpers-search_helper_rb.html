454   def shorten_url (url, length=30)
 
455     return url if url.length <= length
 
456     if url.count('/') >= 4
 
457       if url.index('?')
 
458         arr = url[0..url.index('?')].split('/')
 
459         arr[arr.size - 1] += url[url.index('?') + 1..-1]
 
460       else
 
461         arr = url.split('/')
 
462       end
 
463       host = arr[0]+"//"+arr[2]
 
464       if arr.last.index('?')
 
465         doc_path = arr.last.split('?')
 
466         doc = [doc_path.first, "?", doc_path[1].split('&').first, "..."].join
 
467       else
 
468         doc = arr.last.split('?').first
 
469       end
 
470       [host, "...", doc].join('/')
 
471     else
 
472       url[0, length]+"..."
 
473     end
 
474   end
